############# Travelers API #############

curl -X GET --location "http://127.0.0.1:8080/travelers
curl -X GET --location "http://127.0.0.1:8080/travelers/100

curl -X POST --location "http://127.0.0.1:8080/travelers" -H "Content-Type: application/json" -d '{"id": "100", "firstName": "Joe", "lastName": "Smith"}'

curl -X DELETE  --location "http://127.0.0.1:8080/travelers/300/delete" -H "Content-Type: application/json"
